# Modified:
# * homicidals are not blanket-denied from assimilating (potential)
# * non-gestalt empires with tech_sapient_ai are allowed to (de)assimilate machine units
# * cybernetic/synthetic assimilation can be restricted (with Civic: Organic Zealots)
# * set up checks that support empires with multiple assimilation types (fully enabling that requires adjusting ascension perks)
citizenship_assimilation = {
	is_assimilation = yes
	potential = {
		OR = {
			has_synthethic_dawn = yes
			has_utopia = yes
		}
		exists = from
		from = {
			OR = {
				AND = {
					is_machine_empire = yes
					OR = {
						has_active_tradition = tr_synthetics_adopt_machine # synthetic assimilation (machine)
						has_valid_civic = civic_machine_assimilator # cybernetic assimilation (driven assimilator)
					}
				}
				AND = {
					is_hive_empire = yes
					NOT = { has_valid_civic = civic_hive_devouring_swarm }
					OR = {
						has_tradition = tr_genetics_creation # hive assimilation
						has_tradition = tr_cybernetics_transubstantiation_synthesis # cybernetic assimilation (hive)
					}
				}
				# non-gestalts
				AND = {
					is_gestalt = no
					NOT = { has_valid_civic = civic_fanatic_purifiers }
					OR = {
						# machine deassimilation
						AND = {
							has_technology = tech_sapient_ai
							prev = { has_trait = trait_machine_unit }
						}
						has_country_flag = synth_assimilation # synthetic assimilation (regular)
						has_active_tradition = tr_cybernetics_transubstantiation_synthesis # cybernetic assimilation (regular)
						has_active_tradition = tr_psionics_psionic_assimilation # psionic assimilation
						has_active_tradition = tr_genetics_creation # hive deassimilation
					}
				}
			}
		}
	}
	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = MUST_NOT_BE_DOMINANT_SPECIES
			NOT = { is_exact_same_species = from }
		}
		if = {
			limit = { is_lonely_hive_mind_pop = yes }
			custom_tooltip = {
				fail_text = lonely_hive_mind_pops_kill_themselves_citizenship_assimilation
				from = {
					OR = {
						has_tradition = tr_genetics_creation
						has_tradition = tr_cybernetics_transubstantiation_synthesis
						has_tradition = tr_cybernetics_assimilator_inevitable_assimilation
					}
				}
			}
		}
		if = {
			limit = { is_disconnected_machine_pop = yes }
			custom_tooltip = {
				fail_text = disconnected_machine_pops_collapse_citizenship
				from = { is_hive_empire = no }
			}
			custom_tooltip = {
				fail_text = disconnected_machine_pops_collapse_citizenship_assimilation
				from = { has_technology = tech_sapient_ai }
			}
		}
		OR = {
			if = {
				limit = {
					has_trait = trait_machine_unit
					from = {
						is_gestalt = no
						has_technology = tech_sapient_ai
					}
				}
				# machine deassimilation
				AND = {
					custom_tooltip = {
						text = machine_deassimilation_robots_must_be_allowed
						from = { has_policy_flag = robots_allowed }
					}
					custom_tooltip = {
						text = machine_deassimilation_ai_must_be_allowed
						from = { NOT = { has_policy_flag = ai_outlawed } }
					}
				}
			}
			if = {
				limit = { from = { has_valid_civic = civic_machine_assimilator } }
				# cybernetic assimilation (driven assimilator)
				if = {
					limit = { from = { NOT = { has_active_tradition = tr_cybernetics_assimilator_inevitable_assimilation } } }
					custom_tooltip = {
						success_text = cybernetic_assimilation_assimilator_species_must_be_non_hive_and_non_cybernetic
						fail_text = cybernetic_assimilation_assimilator_species_must_be_non_hive_and_non_cybernetic_fail
						NOR = {
							has_trait = trait_cybernetic
							has_trait = trait_hive_mind
						}
					}
				}
				else = {
					custom_tooltip = {
						text = cybernetic_assimilation_gestalt_species_must_be_not_hive_or_not_cybernetic
						NAND = {
							has_trait = trait_cybernetic
							has_trait = trait_hive_mind
						}
					}
				}
			}
			if = {
				limit = { from = { has_active_tradition = tr_synthetics_adopt_machine } }
				# synthetic assimilation (machine)
				if = {
					limit = { NOT = { has_valid_civic = civic_machine_terminator } }
					custom_tooltip = {
						text = synthetic_assimilation_machine_must_not_be_organic
						OR = {
							has_trait = trait_cybernetic
							is_robotic_species = yes
						}
					}
				}
				else = {
					custom_tooltip = {
						success_text = synthetic_assimilation_machine_terminator_must_not_be_organic_pass
						fail_text = synthetic_assimilation_machine_terminator_must_not_be_organic_fail
						is_robotic_species = yes
					}
				}
			}
			if = {
				limit = { from = { has_country_flag = synth_assimilation } }
				# synthetic assimilation (regular)
				custom_tooltip = {
					text = synthetic_assimilation_must_not_be_hive
					NOT = { has_trait = trait_hive_mind }
				}
			}
			if = {
				limit = { from = { has_active_tradition = tr_cybernetics_transubstantiation_synthesis } }
				# cybernetic assimilation (hive)
				if = {
					limit = { from = { is_hive_empire = yes } }
					custom_tooltip = {
						text = cybernetic_assimilation_gestalt_species_must_be_not_hive_or_not_cybernetic
						NAND = {
							has_trait = trait_hive_mind
							has_trait = trait_cybernetic
						}
					}
				}
				# cybernetic assimilation (regular)
				else_if = {
					limit = { from = { is_gestalt = no } }
					custom_tooltip = {
						text = cybernetic_assimilation_species_must_be_organic
						is_robotic_species = no
						OR = {
							has_trait = trait_hive_mind
							NOT = { has_trait = trait_cybernetic }
						}
					}
				}
			}
			if = {
				limit = { from = { has_active_tradition = tr_psionics_psionic_assimilation } }
				# psionic assimilation
				AND = {
					custom_tooltip = {
						text = psionic_assimilation_species_must_not_be_robotic
						is_robotic_species = no
					}
					custom_tooltip = {
						text = psionic_assimilation_species_must_not_be_psionic
						NOT = { has_trait = trait_psionic }
					}
					custom_tooltip = {
						text = psionic_assimilation_species_must_not_be_hive
						NOT = { has_trait = trait_hive_mind }
					}
				}
			}
			if = {
				limit = { from = { has_active_tradition = tr_genetics_creation } }
				# hive assimilation
				if = {
					limit = { from = { is_hive_empire = yes } }
					custom_tooltip = {
						text = hive_assimilation_species_must_not_be_hive
						NOT = { has_trait = trait_hive_mind }
					}
				}
				# hive deassimilation
				else_if = {
					limit = { from = { is_gestalt = no } }
					custom_tooltip = {
						text = hive_deassimilation_species_must_be_hive
						has_trait = trait_hive_mind
					}
				}
			}
		}
	}
	ai_will_do = {
		factor = 100
		modifier = {
			factor = 0
			# modified so terminator AIs can assimilate other machines/robots
			from = {
				OR = {
					AND = {
						has_valid_civic = civic_machine_terminator
						NAND = {
							has_active_tradition = tr_synthetics_adopt_machine
							prev = { is_robotic_species = yes }
						}
					}
					has_valid_civic = civic_hive_devouring_swarm
					has_valid_civic = civic_fanatic_purifiers
				}
			}
		}
	}
}
