# edited to disallow (de)assimilating machines full service - notably they generate leaders without the synthetic trait (they would eventually get it as the species is assimilated, but I feel it is bad UX)
# also prohibits necrophages from having non-main-species military leaders, prevents nonsapients or syncretic proles from being military leaders, and allows my special necrophage/xenophobe bypass flags
# does not include the translations or Battle Thrall changes from "Special Leadership Privileges for Battle Thralls and Bio-Trophies" - this will will be overwritten by that one, which properly supports all of these features
military_service_full = {
	potential = {
		if = {
			limit = { from = { is_machine_empire = yes } }
			OR = {
				has_trait = trait_machine_unit
				has_trait = trait_mechanical
				has_trait = trait_cybernetic
				AND = {
					has_global_flag = battle_thrall_military_leaders_installed
					from = { has_valid_civic = civic_machine_servitor }
				}
			}
		}
	}
	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = MACHINE_NO_PURE_ORGANIC_SOLDIERS
			if = {
				limit = { from = { is_machine_empire = yes } }
				OR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
					AND = {
						has_trait = trait_cybernetic
						from = { has_civic = civic_machine_assimilator }
					}
					AND = {
						has_global_flag = battle_thrall_military_leaders_installed
						has_species_flag = allow_leaders_organic_advisors@from
						has_citizenship_type = {
							country = from
							type = citizenship_organic_trophy
						}
						from = { has_valid_civic = civic_machine_servitor }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = is_undesirables_tooltip
			NOR = {
				has_citizenship_type = {
					country = from
					type = citizenship_purge_machine
				}
				has_citizenship_type = {
					country = from
					type = citizenship_purge
				}
			}
		}
		custom_tooltip = {
			fail_text = is_slaves_tooltip
			OR = {
				NOT = {
					has_citizenship_type = {
						country = from
						type = citizenship_slavery
					}
				}
				AND = {
					has_global_flag = battle_thrall_military_leaders_installed
					has_slavery_type = {
						country = from
						type = slavery_military
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = is_servitude_tooltip
			NOT = {
				has_citizenship_type = {
					country = from
					type = citizenship_robot_servitude
				}
			}
		}
		custom_tooltip = {
			fail_text = too_alien
			OR = {
				is_same_species = from
				has_species_flag = allow_military_leaders_honored_protectors@from # surprise flag that will help us later
				from = { is_xenophobe = no }
			}
		}
		custom_tooltip = {
			fail_text = too_alien
			OR = {
				is_same_species = from
				has_species_flag = allow_military_leaders_honored_protectors@from # surprise flag that will help us later
				from = { dm_is_necrophage_empire = no }
			}
		}
		custom_tooltip = {
			fail_text = HIVE_MIND_SPECIES_CITIZENS_ONLY
			OR = {
				from = { is_hive_empire = no }
				has_trait = trait_hive_mind
			}
		}
		custom_tooltip = {
			fail_text = NON_MACHINE_SPECIES_NEED_PREREQ_TECH
			if = {
				limit = {
					OR = {
						has_trait = trait_mechanical
						has_trait = trait_machine_unit
					}
				}
				from = {
					OR = {
						is_machine_empire = yes
						has_technology = tech_synthetic_leaders
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = nerve_stapled_cannot_lead_armies
			dm_is_species_smart_enough_for_leadership = yes
		}
		custom_tooltip = {
			fail_text = SPECIES_MUST_NOT_BE_ASSIMILATING
			if = {
				limit = { has_global_flag = deassimilate_machines_installed }
				NAND = {
					has_trait = trait_machine_unit
					has_living_standard = {
						country = from
						type = living_standard_deassimilation
					}
				}
			}
		}
	}
	triggered_pop_modifier = {
		potential = {
			has_global_flag = battle_thrall_military_leaders_installed
			has_slavery_type = { type = slavery_military }
		}
		modifier = {
			pop_cat_slave_happiness = 0.1
		}
	}
	ai_will_do = {
		factor = 5
		modifier = {
			factor = 0
			from = { has_valid_ai_personality = no }
		}
		modifier = {
			factor = 0
			has_global_flag = battle_thrall_military_leaders_installed
			from = { has_valid_civic = civic_machine_servitor }
			NOR = {
				has_trait = trait_mechanical
				has_trait = trait_machine_unit
			}
			dm_is_species_smart_enough_for_leadership = no
			# note: does not include the more complex criteria from BTML because this version gets overwritten anyway
		}
		modifier = {
			factor = 0
			NOT = { has_species_flag = allow_military_leaders_honored_protectors@from }
			from = {
				has_valid_ai_personality = yes
				has_ai_personality_behaviour = multispecies
			}
		}
	}
}
