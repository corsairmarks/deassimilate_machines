# Modified: allow machine deassimilation
living_standard_deassimilation = {
	token = living_standard_deassimilation
	name = living_standard_deassimilation
	pop_modifier = {}
	potential = {
		has_utopia = yes
		exists = from
		OR = {
			# original condition - hive deassimilation
			AND = {
				is_lonely_hive_mind_pop = yes
				from = { has_ascension_perk = ap_evolutionary_mastery }
			}
			# new condition - machine deassimilation
			AND = {
				is_disconnected_machine_pop = yes
				from = { has_technology = tech_sapient_ai }
			}
		}
	}
	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = SPECIES_MUST_BE_ASSIMILATING
			has_citizenship_type = {
				type = citizenship_assimilation
				country = from
			}
		}
		if = {
			limit = {
				has_global_flag = civic_organic_zealots_installed
				is_disconnected_machine_pop = yes
				from = { has_technology = tech_sapient_ai }
			}
			custom_tooltip = {
				fail_text = lost_robot_policy_war_fail
				from = { NOT = { has_country_flag = lost_robot_policy_war } }
			}
			custom_tooltip = {
				fail_text = is_overlord_organic_zealots_fail
				from = {
					NAND = {
						exists = overlord
						overlord = { has_valid_civic = civic_organic_zealots }
					}
				}
			}
		}
	}
	ai_will_do = {
		factor = 100
	}
}
